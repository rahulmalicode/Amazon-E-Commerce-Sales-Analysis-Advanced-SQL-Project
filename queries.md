 as

  DELIMITER //

CREATE PROCEDURE add_sale(
    IN p_product_id INT,
    IN p_qty_sold INT
)
BEGIN
    DECLARE v_stock INT;

    -- Get current stock
    SELECT stock_qty
    INTO v_stock
    FROM inventory
    WHERE product_id = p_product_id;

    -- Check stock availability
    IF v_stock >= p_qty_sold THEN

        -- Insert sales record
        INSERT INTO sales(product_id, qty_sold, sale_date)
        VALUES (p_product_id, p_qty_sold, CURDATE());

        -- Update inventory
        UPDATE inventory
        SET stock_qty = stock_qty - p_qty_sold
        WHERE product_id = p_product_id;

    ELSE
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Insufficient Stock';
    END IF;

END //

DELIMITER ;
